function updateScoreLabel() { const label = document.getElementById('scoreLabel'); let pointValue = parseInt(document.getElementById('pointValue').value, 10); if (isNaN(pointValue)) { label.textContent = "Score: 0"; return; } const scoreMap = { 1: "One Pair", 2: "Two Pair", 3: "Three of a Kind", 4: "Straight", 5: "Flush", 6: "Full House", 7: "Four of a Kind", 8: "Straight Flush", 20: "Royal Flush", 15: "Chicago", -15: "Better Luck Next Time" }; if (pointValue < -15 || pointValue > 20) { label.textContent = "Invalid Score: Please enter a value between -15 and 20."; } else { label.textContent = scoreMap[pointValue] || `Score: ${pointValue}`; } }
